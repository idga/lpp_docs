def indice_energia

        v = self.vct
        
        if v < 670
                return 1
        elsif v >= 670 && v <= 830
                return 2
        else
                return 3
        end
end

def pcarbohidratos
        cantidad = 0
        total = 0

        @gramos.each do |i|
                total += i
        end

        @alimentos.each do |j|
                cantidad += j.carbohidratos
        end

        (cantidad/total)*100
end
